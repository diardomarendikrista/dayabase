# Analisis Lowongan Kerja
SELECT j.title, COUNT(a.id) AS total_applicants
FROM jobs j
LEFT JOIN applications a ON j.id = a.job_id
GROUP BY j.title;


# Statistik Pelamar 
SELECT education, COUNT(*) FROM applicants GROUP BY education;

# Status Aplikasi
SELECT status, COUNT(*) FROM applications GROUP BY status;

# Evaluasi Tes
SELECT t.test_name, AVG(tr.score) AS avg_score
FROM test_results tr
JOIN tests t ON tr.test_id = t.id
GROUP BY t.test_name;

# Gabungan Multi-Tabel
SELECT 
  a.name,
  j.title,
  t.test_name,
  tr.score,
  tr.passed
FROM applicants a
JOIN applications ap ON a.id = ap.applicant_id
JOIN jobs j ON ap.job_id = j.id
JOIN test_results tr ON ap.id = tr.application_id
JOIN tests t ON tr.test_id = t.id
WHERE j.title = 'Backend Developer';
